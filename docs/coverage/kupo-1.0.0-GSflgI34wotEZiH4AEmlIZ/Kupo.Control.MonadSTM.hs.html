   17 -- | An exception-safe bracket-style acquisition of a TMVar.
   18 withTMVar :: (MonadSTM m, MonadMask m) => TMVar m a -> (a -> m b) -> m b
   19 withTMVar var action = mask $ \restore -> do
   20     a <- atomically (takeTMVar var)
   21     b <- restore (action a) `onException` atomically (putTMVar var a)
   22     b <$ atomically (putTMVar var a)
