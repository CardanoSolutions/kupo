   70 runServer
   71     :: Tracer IO TraceHttpServer
   72     -> (forall a. (Database IO -> IO a) -> IO a)
   73     -> IO Health
   74     -> String
   75     -> Int
   76     -> IO ()
   77 runServer tr withDatabase readHealth host port =
   78     Warp.runSettings settings $ tracerMiddleware tr (app withDatabase readHealth)
   79   where
   80     settings = Warp.defaultSettings
   81         & Warp.setPort port
   82         & Warp.setHost (fromString host)
   83         & Warp.setServerName "kupo"
   84         & Warp.setBeforeMainLoop (logWith tr TraceServerListening{host,port})

   86 --
   87 -- Router
   88 --
   89 
   90 app
   91     :: (forall a. (Database IO -> IO a) -> IO a)
   92     -> IO Health
   93     -> Application
   94 app withDatabase readHealth req send =
   95     case (requestMethod req, pathInfo req) of
   96         ("GET", [ "v1", "health" ]) ->
   97             send =<< (handleGetHealth <$> readHealth)
   98 
   99         ("GET", [ "v1", "checkpoints" ]) ->
  100             withDatabase (send . handleGetCheckpoints)
  101 
  102         ("GET", [ "v1", "matches" ]) ->
  103             withDatabase (send . handleGetMatches Nothing)
  104 
  105         ("GET", [ "v1", "matches", arg0 ]) ->
  106             withDatabase (send . handleGetMatches (Just (arg0, Nothing)))
  107 
  108         ("GET", [ "v1", "matches", arg0, arg1 ]) ->
  109             withDatabase (send . handleGetMatches (Just (arg0, Just arg1)))
  110 
  111         ("GET", _) ->
  112             send handleNotFound
  113 
  114         (_, _) ->
  115             send handleMethodNotAllowed
  116 
  117 --
  118 -- Handlers
  119 --
  120 
  121 handleGetHealth
  122     :: Health
  123     -> Response
  124 handleGetHealth =
  125     responseJson status200 defaultHeaders
  126 
  127 handleGetCheckpoints
  128     :: Database IO
  129     -> Response
  130 handleGetCheckpoints Database{..} = do
  131     responseStreamJson (pointToJson @StandardCrypto) $ \yield done -> do
  132         points <- runTransaction (listCheckpointsDesc pointFromRow)
  133         mapM_ yield points
  134         done
  135 
  136 handleGetMatches
  137     :: Maybe (Text, Maybe Text)
  138     -> Database IO
  139     -> Response
  140 handleGetMatches query Database{..} = do
  141     let txt = maybe wildcard (\(a0, a1) -> a0 <> maybe "" ("/" <>) a1) query
  142     case patternFromText @StandardCrypto txt of
  143         Nothing ->
  144             handleInvalidPattern
  145         Just p -> do
  146             responseStreamJson (resultToJson @StandardCrypto) $ \yield done -> do
  147                 runTransaction $ foldInputsByAddress
  148                     (patternToQueryLike p)
  149                     (yield . resultFromRow)
  150                 done
  151 
  152 handleInvalidPattern :: Response
  153 handleInvalidPattern = do
  154     responseJson status400 defaultHeaders $ HttpError
  155         { hint = "Invalid pattern! To fetch matches, you may provide any valid \
  156                  \pattern, including wildcards ('*') or full addresses. Make \
  157                  \sure to double-check the documentation at: \
  158                  \<https://cardanosolutions.github.io/kupo>!"
  159         }
  160 
  161 handleNotFound :: Response
  162 handleNotFound =
  163     responseJson status404 defaultHeaders $ HttpError
  164         { hint = "Endpoint not found. Make sure to double-check the \
  165                  \documentation at: <https://cardanosolutions.github.io/kupo>!"
  166         }
  167 
  168 handleMethodNotAllowed :: Response
  169 handleMethodNotAllowed =
  170     responseJson status406 defaultHeaders $ HttpError
  171         { hint = "Unsupported method called on known endpoint. Make sure to \
  172                  \double-check the documentation at: \
  173                  \<https://cardanosolutions.github.io/kupo>!"
  174         }

  180 -- | Performs a health check against a running server, this is a standalone
  181 -- program which exits immediately, either with a success or an error code.
  182 healthCheck :: String -> Int -> IO ()
  183 healthCheck host port = do
  184     response <- handle onAnyException $ join $ httpLbs
  185         <$> parseRequest ("http://" <> host <> ":" <> show port <> "/v1/health")
  186         <*> newManager defaultManagerSettings
  187     case Json.decode (responseBody response) >>= getConnectionStatus of
  188         Just st | Json.value st == Json.toEncoding Connected ->
  189             return ()
  190         _ -> do
  191             exitWith (ExitFailure 1)
  192   where
  193     onAnyException (_ :: SomeException) =
  194         exitWith (ExitFailure 1)
  195     getConnectionStatus =
  196         lookup @(Map String Json.Value) "connection_status"

  198 --
  199 -- Helpers
  200 --
  201 
  202 data HttpError = HttpError
  203     { hint :: Text }
  204     deriving stock (Generic)
  205     deriving anyclass (ToJSON)
  206 
  207 defaultHeaders :: [Header]
  208 defaultHeaders =
  209     [ ( hContentType, "application/json;charset=utf-8" )
  210     ]
  211 
  212 responseJson
  213     :: ToJSON a
  214     => Http.Status
  215     -> [Header]
  216     -> a
  217     -> Response
  218 responseJson status headers a =
  219     let
  220         bytes = B.toLazyByteString $ Json.fromEncoding (Json.toEncoding a)
  221         len = BL.length bytes
  222         contentLength = ( hContentLength, encodeUtf8 (show @Text len) )
  223      in
  224         responseLBS status (contentLength : headers) bytes
  225 
  226 responseStreamJson
  227     :: (a -> Json.Encoding)
  228     -> ((a -> IO ()) -> IO () -> IO ())
  229     -> Response
  230 responseStreamJson encode callback = do
  231     responseStream status200 defaultHeaders $ \write flush -> do
  232         ref <- newIORef True
  233         write openBracket
  234         callback
  235             (\a -> do
  236                 isFirstResult <- readIORef ref
  237                 write (separator isFirstResult <> Json.fromEncoding (encode a))
  238                 writeIORef ref False
  239             )
  240             (write closeBracket >> flush)
  241   where
  242     openBracket = B.putCharUtf8 '['
  243     closeBracket = B.putCharUtf8 ']'
  244     separator isFirstResult
  245         | isFirstResult = mempty
  246         | otherwise     = B.putCharUtf8 ','

  249 --
  250 -- Tracer
  251 --
  252 
  253 tracerMiddleware :: Tracer IO TraceHttpServer -> Middleware
  254 tracerMiddleware tr runApp req send = do
  255     runApp req $ \res -> do
  256         let status = mkStatus (responseStatus res)
  257         logWith tr $ TraceRequest {method, path, status}
  258         send res
  259   where
  260     method = decodeUtf8 (requestMethod req)
  261     path = pathInfo req
  262 
  263 data TraceHttpServer where
  264     TraceServerListening
  265         :: { host :: String, port :: Int }
  266         -> TraceHttpServer
  267     TraceRequest
  268         :: { method :: Text, path :: [Text], status :: Status }
  269         -> TraceHttpServer
  270     deriving stock (Generic)
  271 
  272 instance HasSeverityAnnotation TraceHttpServer where
  273     getSeverityAnnotation = \case
  274         TraceServerListening{} -> Notice
  275         TraceRequest{} -> Info
  276 
  277 instance ToJSON TraceHttpServer where
  278     toEncoding =
  279         defaultGenericToEncoding

  285 data Status = Status
  286     { statusCode :: Int
  287     , statusMessage :: Text
  288     } deriving stock (Generic)
  289       deriving anyclass (ToJSON)
  290 
  291 mkStatus :: Http.Status -> Status
  292 mkStatus status = Status
  293     { statusCode = Http.statusCode status
  294     , statusMessage = decodeUtf8 (Http.statusMessage status)
  295     }
