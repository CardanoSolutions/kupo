    5 -- | A simple eDSL on top of TBQueue to have a concurrent producer / consumer
    6 -- interface to decouple two processes. The main hypotheseses are:
    7 --
    8 -- - The producer produces message one-by-one
    9 -- - The produces produces message a lot faster than the consumer can consome
   10 --   then sequentially
   11 -- - The consumer can however consume multiple messages at once.
   12 module Kupo.App.Mailbox
   13     ( Mailbox
   14     , newMailbox
   15     , putMailbox
   16     , flushMailbox
   17     ) where
   18 
   19 import Kupo.Prelude
   20 
   21 import Kupo.Control.MonadSTM
   22     ( MonadSTM (..) )
   23 
   24 type Mailbox m a = TBQueue m a
   25 
   26 newMailbox
   27     :: (MonadSTM m)
   28     => Natural
   29         -- ^ Size of the mailbox.
   30     -> STM m (Mailbox m a)
   31 newMailbox =
   32     newTBQueue
   33 {-# INLINEABLE newMailbox #-}
   34 
   35 putMailbox
   36     :: (MonadSTM m)
   37     => Mailbox m a
   38     -> a
   39     -> STM m ()
   40 putMailbox =
   41     writeTBQueue
   42 {-# INLINEABLE putMailbox #-}
   43 
   44 flushMailbox
   45     :: (MonadSTM m)
   46     => Mailbox m a
   47     -> STM m (NonEmpty a)
   48 flushMailbox q = do
   49     h <- readTBQueue q
   50     rest <- flushTBQueue q
   51     pure (h :| rest)
   52 {-# INLINEABLE flushMailbox #-}
