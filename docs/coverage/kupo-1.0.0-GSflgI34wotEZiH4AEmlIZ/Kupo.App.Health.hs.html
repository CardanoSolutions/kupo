   23 -- | Safe and limited accessor to the health
   24 readHealth
   25     :: forall m.
   26         ( MonadSTM m
   27         )
   28     => TVar m Health
   29     -> m Health
   30 readHealth =
   31     atomically . readTVar

   33 -- | Creates an isolated effectful function to toggle a health connection status.
   34 connectionStatusToggle
   35     :: forall m.
   36         ( MonadSTM m
   37         )
   38     => TVar m Health
   39     -> ConnectionStatusToggle m
   40 connectionStatusToggle health =
   41     ConnectionStatusToggle
   42         { toggleConnected = setConnectionStatus Connected
   43         , toggleDisconnected = setConnectionStatus Disconnected
   44         }
   45   where
   46     setConnectionStatus connectionStatus =
   47         atomically (modifyTVar' health (\h -> h { connectionStatus }))
   48 
   49 -- | A safe setter for the most recent tip and checkpoint.
   50 recordCheckpoint
   51     :: forall m crypto.
   52         ( MonadSTM m
   53         )
   54     => TVar m Health
   55     -> Tip (Block crypto)
   56     -> Maybe SlotNo
   57     -> m ()
   58 recordCheckpoint health (Just . getTipSlotNo -> mostRecentNodeTip) mostRecentCheckpoint =
   59     atomically $ modifyTVar' health $ \h -> h
   60         { mostRecentNodeTip
   61         , mostRecentCheckpoint
   62         }
