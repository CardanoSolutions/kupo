   99 class MonadOuroboros (m :: Type -> Type) where

  102         :: IsBlock (Block m)

  110         -> m ()

  122 -- | Exception thrown when creating a chain-sync client from an invalid list of
  123 -- points.
  124 data IntersectionNotFoundException = IntersectionNotFoundException
  125     { requestedPoints :: [WithOrigin SlotNo]
  126         -- ^ Provided points for intersection.
  127     , tip :: WithOrigin SlotNo
  128         -- ^ Current known tip of the chain.
  129     } deriving (Show)
  130 instance Exception IntersectionNotFoundException
  131 
  132 instance MonadOuroboros IO where
  133     type Block IO = CardanoBlock StandardCrypto
  134     withChainSyncServer tr ConnectionStatusToggle{..} wantedVersions networkMagic slotsPerEpoch socket client =
  135         withIOManager $ \iocp -> do
  136             connectTo (mkLocalSnocket iocp) tracers versions socket
  137                 & onExceptions
  138                 & foreverCalmly
  139       where
  140         tracers = NetworkConnectTracers
  141             { nctMuxTracer = nullTracer
  142             , nctHandshakeTracer = nullTracer
  143             }
  144 
  145         mkLocalSnocket iocp =
  146             let snocket = localSnocket iocp
  147              in snocket
  148                     { toBearer = \time trMux fd -> do
  149                         bearer <- toBearer snocket time trMux fd
  150                         toggleConnected $> bearer
  151                     }
  152 
  153         versions = combineVersions
  154             [ simpleSingletonVersions v vData (mkOuroborosApplication v)
  155             | v <- wantedVersions
  156             ]
  157           where
  158             vData  = NodeToClientVersionData networkMagic
  159 
  160         mkOuroborosApplication version =
  161             OuroborosApplication $ \_connectionId _controlMessageSTM ->
  162                 [ MiniProtocol
  163                     { miniProtocolNum =
  164                         MiniProtocolNum 5
  165                     , miniProtocolLimits =
  166                         MiniProtocolLimits (fromIntegral $ maxBound @Word32)
  167                     , miniProtocolRun =
  168                         InitiatorProtocolOnly $ MuxPeerRaw $ \channel ->
  169                             let
  170                                 peer = chainSyncClientPeerPipelined client
  171                                 codec = cChainSyncCodec (codecs slotsPerEpoch version)
  172                              in
  173                                 runPipelinedPeer nullTracer codec channel peer
  174                     }
  175                 ]
  176 
  177         foreverCalmly a = do
  178             let a' = a *> threadDelay 5 *> a' in a'
  179 
  180         onExceptions
  181             = handle onUnknownException
  182             . handle onIOException
  183             . (`onException` toggleDisconnected)
  184 
  185         onIOException :: IOException -> IO ()
  186         onIOException e
  187             | isRetryable = do
  188                 traceWith tr $ ChainSyncFailedToConnect socket 5
  189             | otherwise = do
  190                 traceWith tr $ ChainSyncUnknownException $ show (toException e)
  191           where
  192             isRetryable :: Bool
  193             isRetryable = isResourceVanishedError e || isDoesNotExistError e || isTryAgainError e
  194 
  195             isTryAgainError :: IOException -> Bool
  196             isTryAgainError = isInfixOf "resource exhausted" . show
  197 
  198             isResourceVanishedError :: IOException -> Bool
  199             isResourceVanishedError = isInfixOf "resource vanished" . show
  200 
  201         onUnknownException :: SomeException -> IO ()
  202         onUnknownException e
  203             | isAsyncException e = do
  204                 throwIO e
  205             | otherwise = case fromException e of
  206                 Just (_ :: IntersectionNotFoundException) ->
  207                     throwIO e
  208                 Nothing ->
  209                     traceWith tr $ ChainSyncUnknownException $ show e

  211 codecs
  212     :: EpochSlots
  213     -> NodeToClientVersion
  214     -> ClientCodecs (Block IO) IO
  215 codecs epochSlots nodeToClientV =
  216     clientCodecs cfg (supportedVersions ! nodeToClientV) nodeToClientV
  217   where
  218     supportedVersions =
  219         supportedNodeToClientVersions (Proxy @(Block IO))
  220     cfg =
  221         CardanoCodecConfig byron shelley allegra mary alonzo
  222       where
  223         byron   = ByronCodecConfig epochSlots
  224         shelley = ShelleyCodecConfig
  225         allegra = ShelleyCodecConfig
  226         mary    = ShelleyCodecConfig
  227         alonzo  = ShelleyCodecConfig

  233 data TraceChainSync where
  234     ChainSyncRollBackward
  235         :: { point :: SlotNo }
  236         -> TraceChainSync
  237     ChainSyncRollForward
  238         :: { slotNo :: SlotNo, matches :: Int }
  239         -> TraceChainSync
  240     ChainSyncIntersectionNotFound
  241         :: { points :: [WithOrigin SlotNo] }
  242         -> TraceChainSync
  243     ChainSyncFailedToConnect
  244         :: { socket :: FilePath, retryingIn :: DiffTime }
  245         -> TraceChainSync
  246     ChainSyncUnknownException
  247         :: { exception :: Text }
  248         -> TraceChainSync
  249     deriving stock (Generic, Show)
  250 
  251 instance ToJSON TraceChainSync where
  252     toEncoding =
  253         defaultGenericToEncoding
  254 
  255 instance ToJSON (WithOrigin SlotNo) where
  256     toEncoding = \case
  257         Origin -> toEncoding ("origin" :: Text)
  258         At sl -> toEncoding sl
  259 
  260 instance HasSeverityAnnotation TraceChainSync where
  261     getSeverityAnnotation = \case
  262         ChainSyncRollForward{} ->
  263             Debug
  264         ChainSyncRollBackward{} ->
  265             Notice
  266         ChainSyncFailedToConnect{} ->
  267             Warning
  268         ChainSyncIntersectionNotFound{} ->
  269             Error
  270         ChainSyncUnknownException{} ->
  271             Error
