   61 data Command (f :: Type -> Type)
   62     = Run (f NetworkParameters) Configuration (Tracers' IO MinSeverities)
   63     | HealthCheck String Int
   64     | Version
   65 
   66 deriving instance Eq (f NetworkParameters) => Eq (Command f)
   67 deriving instance Show (f NetworkParameters) => Show (Command f)
   68 
   69 parseOptions :: IO (Command Identity)
   70 parseOptions =
   71     customExecParser (prefs showHelpOnEmpty) parserInfo >>= \case
   72         Version -> pure Version
   73         HealthCheck host port -> pure (HealthCheck host port)
   74         Run _ cfg@Configuration{nodeConfig} tracers -> do
   75             networkParameters <- parseNetworkParameters nodeConfig
   76             pure $ Run (Identity networkParameters) cfg tracers
   77 
   78 parseOptionsPure :: [String] -> Either String (Command Proxy)
   79 parseOptionsPure args =
   80     case execParserPure defaultPrefs parserInfo args of
   81         Success a -> Right a
   82         Failure e -> Left (show e)
   83         CompletionInvoked{} -> Left "Completion Invoked."

  272     flag' Version (mconcat
  273         [ long "version"
  274         , short 'v'
  275         , help helpText

  278     subparser (mconcat
  279         [ hidden

  309 parseNetworkParameters :: FilePath -> IO NetworkParameters
  310 parseNetworkParameters configFile = runOrDie $ do
  311     config <- decodeYaml @Yaml.Value configFile
  312     let genesisFiles = (,)
  313             <$> config ^? key "ByronGenesisFile" . _String
  314             <*> config ^? key "ShelleyGenesisFile" . _String
  315     case genesisFiles of
  316         Nothing ->
  317             throwE "Missing 'ByronGenesisFile' and/or 'ShelleyGenesisFile' from \
  318                    \Cardano's configuration (i.e. '--node-config' option)?"
  319         Just (toString -> byronGenesisFile, toString -> shelleyGenesisFile) -> do
  320             byronGenesis   <- decodeYaml (replaceFileName configFile byronGenesisFile)
  321             shelleyGenesis <- decodeYaml (replaceFileName configFile shelleyGenesisFile)
  322             case Json.fromJSON (Json.Object (byronGenesis <> shelleyGenesis)) of
  323                 Json.Error e -> throwE e
  324                 Json.Success params -> pure params
  325   where
  326     runOrDie :: ExceptT String IO a -> IO a
  327     runOrDie = runExceptT >=> either (die . ("Failed to parse network parameters: " <>)) pure
  328 
  329     prettyParseException :: Yaml.ParseException -> String
  330     prettyParseException e = "Failed to decode JSON (or YAML) file: " <> show e
  331 
  332     decodeYaml :: FromJSON a => FilePath -> ExceptT String IO a
  333     decodeYaml = withExceptT prettyParseException . ExceptT . Yaml.decodeFileEither

  339 longline :: Text -> Doc
  340 longline = fillSep . fmap (text . toString) . words

  342 severities :: [String]
  343 severities =
  344     show @_ @Severity <$> [minBound .. maxBound]

  346 readSeverityM :: ReadM (Maybe Severity)
  347 readSeverityM = maybeReader $ \case
  348     [] -> Nothing
  349     (toUpper -> h):q ->
  350         if h:q == "Off" then
  351             Just Nothing
  352         else
  353             Just <$> readMay (h:q)
