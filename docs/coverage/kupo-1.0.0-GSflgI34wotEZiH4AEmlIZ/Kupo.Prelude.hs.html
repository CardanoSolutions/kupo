  107 -- | The runtime does not let the application terminate gracefully when a
  108 -- SIGTERM is received. It does however for SIGINT which allows the application
  109 -- to cleanup sub-processes.
  110 --
  111 -- This function install handlers for SIGTERM and turn them into SIGINT.
  112 hijackSigTerm :: MonadIO m => m ()
  113 hijackSigTerm =
  114     liftIO $ void (installHandler softwareTermination handler empty)
  115   where
  116     handler = CatchOnce (raiseSignal keyboardSignal)

  118 -- | An unsafe version of 'decodeBase16'. Use with caution.
  119 unsafeDecodeBase16 :: HasCallStack => Text -> ByteString
  120 unsafeDecodeBase16 =
  121     either error identity . decodeBase16 . encodeUtf8
  122 
  123 -- | Decode a byte string from 'Base58', re-defining here to align interfaces
  124 -- with base16 & base64.
  125 decodeBase58 :: ByteString -> Either Text ByteString
  126 decodeBase58 =
  127     maybe (Left msg) Right . Base58.decodeBase58 Base58.bitcoinAlphabet
  128   where
  129     msg = "failed to decode Base58-encoded string."
  130 
  131 -- | Encode some byte string to 'Text' as Base58. See note on 'decodeBase58'.
  132 encodeBase58 :: ByteString -> Text
  133 encodeBase58 =
  134     decodeUtf8. Base58.encodeBase58 Base58.bitcoinAlphabet
  135 
  136 -- | Encode a ByteString to JSON as a base16 text
  137 byteStringToJson :: ByteString -> Json.Encoding
  138 byteStringToJson =
  139     Json.text . encodeBase16
  140 
  141 -- | Generic JSON encoding, with pre-defined default options.
  142 defaultGenericToEncoding :: (Generic a, GToJSON' Encoding Zero (Rep a)) => a -> Json.Encoding
  143 defaultGenericToEncoding =
  144     genericToEncoding Json.defaultOptions
