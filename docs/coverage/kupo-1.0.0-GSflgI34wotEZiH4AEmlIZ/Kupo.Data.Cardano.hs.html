  181 foldBlock
  182     :: forall crypto b.
  183         ( Crypto crypto
  184         )
  185     => (Transaction crypto -> b -> b)
  186     -> b
  187     -> Block crypto
  188     -> b
  189 foldBlock fn b = \case
  190     BlockByron{} ->
  191         b
  192     BlockShelley (ShelleyBlock (Ledger.Block _ txs) _) ->
  193         foldr (fn . TransactionShelley) b (Ledger.Shelley.txSeqTxns' txs)
  194     BlockAllegra (ShelleyBlock (Ledger.Block _ txs) _) ->
  195         foldr (fn . TransactionAllegra) b (Ledger.Shelley.txSeqTxns' txs)
  196     BlockMary (ShelleyBlock (Ledger.Block _ txs) _) ->
  197         foldr (fn . TransactionMary) b (Ledger.Shelley.txSeqTxns' txs)
  198     BlockAlonzo (ShelleyBlock (Ledger.Block _ txs) _) ->
  199         foldr (fn . TransactionAlonzo) b (Ledger.Alonzo.txSeqTxns txs)

  253 data Transaction crypto
  254     = TransactionShelley
  255         (Ledger.Shelley.Tx (ShelleyEra crypto))
  256     | TransactionAllegra
  257         (Ledger.Shelley.Tx (AllegraEra crypto))
  258     | TransactionMary
  259         (Ledger.Shelley.Tx (MaryEra crypto))
  260     | TransactionAlonzo
  261         (Ledger.Alonzo.ValidatedTx (AlonzoEra crypto))
  262 
  263 mapMaybeOutputs
  264     :: forall a crypto. (Crypto crypto)
  265     => (OutputReference crypto -> Output crypto -> Maybe a)
  266     -> Transaction crypto
  267     -> [a]
  268 mapMaybeOutputs fn = \case
  269     TransactionShelley tx ->
  270         let
  271             body = Ledger.Shelley.body tx
  272             txId = Ledger.txid @(ShelleyEra crypto) body
  273             outs = Ledger.Shelley._outputs body
  274          in
  275             traverseAndTransform (asAlonzoOutput inject) txId 0 outs
  276     TransactionAllegra tx ->
  277         let
  278             body = Ledger.Shelley.body tx
  279             txId = Ledger.txid @(AllegraEra crypto) body
  280             outs = Ledger.MaryAllegra.outputs' body
  281          in
  282             traverseAndTransform (asAlonzoOutput inject) txId 0 outs
  283     TransactionMary tx ->
  284         let
  285             body = Ledger.Shelley.body tx
  286             txId = Ledger.txid @(MaryEra crypto) body
  287             outs = Ledger.MaryAllegra.outputs' body
  288          in
  289             traverseAndTransform (asAlonzoOutput identity) txId 0 outs
  290     TransactionAlonzo tx ->
  291         let
  292             body = Ledger.Alonzo.body tx
  293             txId = Ledger.txid @(AlonzoEra crypto) body
  294             outs = Ledger.Alonzo.outputs' body
  295          in
  296             traverseAndTransform identity txId 0 outs
  297   where
  298     traverseAndTransform
  299         :: forall output. ()
  300         => (output -> Output crypto)
  301         -> TransactionId crypto
  302         -> Natural
  303         -> StrictSeq output
  304         -> [a]
  305     traverseAndTransform transform txId ix = \case
  306         Empty -> []
  307         output :<| rest ->
  308             let
  309                 outputRef = Ledger.TxIn txId ix
  310                 results   = traverseAndTransform transform txId (succ ix) rest
  311              in
  312                 case fn outputRef (transform output) of
  313                     Nothing ->
  314                         results
  315                     Just result ->
  316                         result : results

  407 unsafeValueFromList
  408     :: forall crypto.
  409         ( HasCallStack
  410         , Crypto crypto
  411         )
  412     => Integer
  413     -> [(ByteString, ByteString, Integer)]
  414     -> Value crypto
  415 unsafeValueFromList ada assets =
  416     Ledger.valueFromList
  417         ada
  418         [ ( unsafePolicyId pid, unsafeAssetName name, q)
  419         | (pid, name, q) <- assets
  420         ]
  421   where
  422     unsafePolicyId =
  423         Ledger.PolicyID
  424         . Ledger.ScriptHash
  425         . UnsafeHash
  426         . toShort
  427         . sizeInvariant (== (digestSize @Blake2b_224))
  428 
  429     unsafeAssetName =
  430         Ledger.AssetName
  431         . sizeInvariant (<= assetNameMaxLength)
  432 
  433 valueToJson :: forall crypto. (Crypto crypto) => Value crypto -> Json.Encoding
  434 valueToJson (Ledger.Value coins assets) = Json.pairs $ mconcat
  435     [ Json.pair "coins"  (Json.integer coins)
  436     , Json.pair "assets" (assetsToJson assets)
  437     ]
  438   where
  439     assetsToJson :: Map (Ledger.PolicyID crypto) (Map Ledger.AssetName Integer) -> Json.Encoding
  440     assetsToJson =
  441         Json.pairs
  442         .
  443         Map.foldrWithKey
  444             (\k v r -> Json.pair (assetIdToText k) (Json.integer v) <> r)
  445             mempty
  446         .
  447         flatten
  448 
  449     flatten :: (Ord k1, Ord k2) => Map k1 (Map k2 a) -> Map (k1, k2) a
  450     flatten = Map.foldrWithKey
  451         (\k inner -> Map.union (Map.mapKeys (k,) inner))
  452         mempty
  453 
  454     assetIdToText :: (Ledger.PolicyID crypto, Ledger.AssetName) -> Text
  455     assetIdToText (Ledger.PolicyID (Ledger.ScriptHash (UnsafeHash pid)), Ledger.AssetName bytes)
  456         | BS.null bytes = encodeBase16 (fromShort pid)
  457         | otherwise     = encodeBase16 (fromShort pid) <> "." <> encodeBase16 bytes

  463 addressToJson :: Address crypto -> Json.Encoding
  464 addressToJson = \case
  465     addr@Ledger.AddrBootstrap{} ->
  466         (Json.text . encodeBase58 . addressToBytes) addr
  467     addr@(Ledger.Addr network _ _) ->
  468         (Json.text . encodeBech32 (hrp network) . addressToBytes) addr
  469   where
  470     hrp = \case
  471         Ledger.Mainnet -> HumanReadablePart "addr"
  472         Ledger.Testnet -> HumanReadablePart "addr_test"

  479 addressFromBytes = Ledger.deserialiseAddr
  480 {-# INLINEABLE addressFromBytes #-}

  581 sizeInvariant :: HasCallStack => (Int -> Bool) -> ByteString -> ByteString
  582 sizeInvariant predicate bytes
  583     | predicate (BS.length bytes) =
  584         bytes
  585     | otherwise =
  586         error ("predicate failed for bytes: " <> show bytes)
