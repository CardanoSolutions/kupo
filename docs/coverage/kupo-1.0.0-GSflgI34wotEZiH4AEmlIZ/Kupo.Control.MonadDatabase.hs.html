  126 --
  127 -- IO

  129 
  130 instance MonadDatabase IO where
  131     type DBTransaction IO = ReaderT Connection IO
  132     data DBLock IO = DBLock (TVar IO Word) (TVar IO Bool) (TMVar IO Connection)
  133 
  134     newLock = DBLock <$> newTVarIO 0 <*> newTVarIO True <*> newEmptyTMVarIO
  135 
  136     withDatabase tr mode (DBLock readers writer inMemory) k filePath action
  137         | filePath == ":memory:" = do
  138             traceWith tr DatabaseRunningInMemory
  139             case mode of
  140                 Write -> do
  141                     withConnection filePath $ \conn -> do
  142                         databaseVersion conn >>= runMigrations tr conn
  143                         atomically $ putTMVar inMemory conn
  144                         action (mkDatabase k (bracketConnection conn))
  145                 ReadOnly -> do
  146                     conn <- atomically $ readTMVar inMemory
  147                     action (mkDatabase k (bracketConnection conn))
  148         | otherwise = do
  149             withConnection filePath $ \conn -> do
  150                 when (mode == Write) (databaseVersion conn >>= runMigrations tr conn)
  151                 action (mkDatabase k (bracketConnection conn))
  152       where
  153         -- The heuristic below aims at favoring readers over writers. We assume
  154         -- that there is only one writer and possibly many readers. The count of
  155         -- readers is kept in a TVar and readers increment/decrement it around
  156         -- each transactions.
  157         -- The writer however will only attempt writing if there's currently no
  158         -- reader. In case where there's at least one reader waiting, it'll let
  159         -- them pass and wait until all readers are done.
  160         bracketConnection :: Connection -> (forall a. ((Connection -> IO a) -> IO a))
  161         bracketConnection conn between =
  162             case mode of
  163                 ReadOnly -> bracket_
  164                     (do
  165                         atomically (modifyTVar' readers succ)
  166                         atomically (readTVar writer >>= check . not)
  167                     )
  168                     (atomically (modifyTVar' readers pred))
  169                     (between conn)
  170 
  171                 Write -> bracket_
  172                     (atomically $ do
  173                         readTVar readers >>= check . (== 0)
  174                         writeTVar writer True
  175                     )
  176                     (atomically $ writeTVar writer False)
  177                     (between conn)
  178 
  179 mkDatabase :: LongestRollback -> (forall a. (Connection -> IO a) -> IO a) -> Database IO
  180 mkDatabase (fromIntegral -> longestRollback) bracketConnection = Database
  181     { insertInputs = \inputs -> ReaderT $ \conn ->
  182         mapM_
  183         (\Input{..} ->
  184             insertRow @"inputs" conn
  185                 [ SQLBlob outputReference
  186                 , SQLText address
  187                 , SQLBlob value
  188                 , maybe SQLNull SQLBlob datumHash
  189                 , SQLBlob headerHash
  190                 , SQLInteger (fromIntegral slotNo)
  191                 ]
  192         )
  193         inputs
  194 
  195     , foldInputsByAddress = \addressLike yield -> ReaderT $ \conn -> do
  196         let matchMaybeDatumHash = \case
  197                 SQLBlob datumHash -> Just datumHash
  198                 _ -> Nothing
  199         let qry = "SELECT output_reference, address, value, datum_hash, header_hash, slot_no, LENGTH(address) as len \
  200                   \FROM inputs WHERE address " <> addressLike <> " ORDER BY slot_no DESC"
  201         fold_ conn (Query qry) () $ \() -> \case
  202             [ SQLBlob outputReference
  203                 , SQLText address
  204                 , SQLBlob value
  205                 , matchMaybeDatumHash -> datumHash
  206                 , SQLBlob headerHash
  207                 , SQLInteger (fromIntegral -> slotNo)
  208                 , _ -- LENGTH(address)
  209                 ] -> yield Input{..}
  210             (xs :: [SQLData]) -> throwIO (UnexpectedRow addressLike [xs])
  211 
  212     , insertCheckpoint = \Checkpoint{..} -> ReaderT $ \conn -> do
  213         insertRow @"checkpoints" conn
  214             [ SQLBlob checkpointHeaderHash
  215             , SQLInteger (fromIntegral checkpointSlotNo)
  216             ]
  217         execute conn "DELETE FROM checkpoints WHERE slot_no < ?"
  218             [ SQLInteger (fromIntegral checkpointSlotNo - longestRollback)
  219             ]
  220 
  221     , listCheckpointsDesc = \mk -> ReaderT $ \conn -> do
  222         -- NOTE: fetching in *ASC*ending order because the list construction
  223         -- reverses it,
  224         fold_ conn "SELECT * FROM checkpoints ORDER BY slot_no ASC" []
  225             $ \xs (checkpointHeaderHash, checkpointSlotNo) ->
  226                 pure ((mk Checkpoint{..}) : xs)
  227 
  228     , rollbackTo = \slotNo -> ReaderT $ \conn -> do
  229         execute conn "DELETE FROM inputs WHERE slot_no > ?"
  230             [ SQLInteger (fromIntegral slotNo)
  231             ]
  232         execute conn "DELETE FROM checkpoints WHERE slot_no > ?"
  233             [ SQLInteger (fromIntegral slotNo)
  234             ]
  235         query_ conn "SELECT MAX(slot_no) FROM checkpoints" >>= \case
  236             [[SQLInteger slotNo']] ->
  237                 return $ Just (fromIntegral slotNo')
  238             [[SQLNull]] ->
  239                 return Nothing
  240             xs ->
  241                 throwIO $ UnexpectedRow ("MAX(" <> show slotNo <> ")") xs
  242 
  243     , runTransaction = \r -> bracketConnection $ \conn ->
  244         withTransaction conn (runReaderT r conn)
  245     }
  246 
  247 insertRow
  248     :: forall tableName.
  249         ( KnownSymbol tableName
  250         )
  251     => Connection
  252     -> [SQLData]
  253     -> IO ()
  254 insertRow conn r =
  255     let
  256         tableName = fromString (symbolVal (Proxy @tableName))
  257         values = mkPreparedStatement (length (toRow r))
  258         qry = "INSERT OR IGNORE INTO " <> tableName <> " VALUES " <> values
  259      in
  260         execute conn qry r
  261

  266 mkPreparedStatement :: Int -> Query
  267 mkPreparedStatement n =
  268     Query ("(" <> T.intercalate "," (replicate n "?") <> ")")

  270 --
  271 -- Migrations
  272 --
  273 
  274 type MigrationRevision = Int
  275 type Migration = [Query]
  276 
  277 databaseVersion :: Connection -> IO MigrationRevision
  278 databaseVersion conn =
  279     withStatement conn "PRAGMA user_version" $ \stmt -> do
  280         nextRow stmt >>= \case
  281             Just (Only version) ->
  282                 pure version
  283             _ ->
  284                 throwIO UnexpectedUserVersion
  285 
  286 runMigrations :: Tracer IO TraceDatabase -> Connection -> MigrationRevision -> IO ()
  287 runMigrations tr conn currentVersion = do
  288     let missingMigrations = drop currentVersion migrations
  289     traceWith tr (DatabaseCurrentVersion currentVersion)
  290     if null missingMigrations then
  291         traceWith tr DatabaseNoMigrationNeeded
  292     else do
  293         traceWith tr $ DatabaseRunningMigration currentVersion (length missingMigrations)
  294         void $ withTransaction conn $
  295             traverse (traverse (execute_ conn)) missingMigrations
  296 
  297 migrations :: [Migration]
  298 migrations =
  299     [ mkMigration ix (decodeUtf8 m)
  300     | (ix, m) <- zip
  301         [1..]
  302         [ $(embedFile "db/001.sql")
  303         , $(embedFile "db/002.sql")
  304         ]
  305     ]
  306   where
  307     mkMigration :: Int -> Text -> [Query]
  308     mkMigration i sql =
  309         ("PRAGMA user_version = " <> show i <> ";")
  310         : (fmap Query . filter (not . T.null . T.strip) . T.splitOn ";") sql
