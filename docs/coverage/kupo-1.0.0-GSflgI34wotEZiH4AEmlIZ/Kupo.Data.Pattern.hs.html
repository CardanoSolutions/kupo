   76 data Pattern crypto
   77     = MatchAny MatchBootstrap
   78     | MatchExact (Address crypto)
   79     | MatchPayment ByteString
   80     | MatchDelegation ByteString
   81     | MatchPaymentAndDelegation ByteString ByteString
   82     deriving (Generic, Eq, Show)
   83 
   84 wildcard :: Text
   85 wildcard = "*"
   86 {-# INLINEABLE wildcard #-}
   87 
   88 patternFromText :: Crypto crypto => Text -> Maybe (Pattern crypto)
   89 patternFromText txt =
   90     readerAny <|> readerExact <|> readerPaymentOrDelegation
   91   where
   92     readerAny = MatchAny IncludingBootstrap
   93         <$ guard (txt == wildcard)
   94 
   95     readerExact =
   96         readerBase16 txt
   97             (\bytes -> do
   98                 MatchExact <$> addressFromBytes bytes
   99             )
  100         <|>
  101         readerBech32 txt
  102             (\bytes hrp -> do
  103                 if | hrp `elem` [ [humanReadablePart|addr|], [humanReadablePart|addr_test|] ] -> do
  104                         MatchExact <$> addressFromBytes bytes
  105                    | hrp `elem` [ [humanReadablePart|stake|], [humanReadablePart|stake_test|] ] -> do
  106                         pure $ MatchDelegation (BS.drop 1 bytes)
  107                    | otherwise -> do
  108                         empty
  109             )
  110         <|>
  111         readerBase58 txt
  112             (\bytes -> do
  113                 MatchExact <$> addressFromBytes bytes
  114             )
  115 
  116     readerPaymentOrDelegation =
  117         case T.splitOn "/" txt of
  118             [payment, delegation] -> do
  119                 if  | payment == wildcard && delegation == wildcard ->
  120                         pure (MatchAny OnlyShelley)
  121                     | payment == wildcard ->
  122                         MatchDelegation
  123                             <$> readerCredential delegation
  124                     | delegation == wildcard ->
  125                         MatchPayment
  126                             <$> readerCredential payment
  127                     | otherwise -> do
  128                         MatchPaymentAndDelegation
  129                             <$> readerCredential payment
  130                             <*> readerCredential delegation
  131             _ ->
  132                 empty
  133 
  134     -- NOTE: byte strings are interpreted either as verification key or
  135     -- verification key hashes depending on their length. In case they are
  136     -- verification key, they're hashed to produce key hashes.
  137     readerCredential str =
  138         readerBase16 str (\bytes -> do
  139             if | BS.length bytes == digestSize @Blake2b_256 -> do
  140                     pure (digest (Proxy @Blake2b_224) bytes)
  141                | BS.length bytes == digestSize @Blake2b_224 -> do
  142                     pure bytes
  143                | otherwise ->
  144                     empty
  145         )
  146         <|>
  147         readerBech32 str (\bytes hrp -> do
  148             if | BS.length bytes == digestSize @Blake2b_256 -> do
  149                     guard $ hrp `elem`
  150                         [ [humanReadablePart|vk|]
  151                         , [humanReadablePart|addr_vk|]
  152                         , [humanReadablePart|stake_vk|]
  153                         ]
  154                     pure (digest (Proxy @Blake2b_224) bytes)
  155                | BS.length bytes == digestSize @Blake2b_224 -> do
  156                     guard $ hrp `elem`
  157                         [ [humanReadablePart|vkh|]
  158                         , [humanReadablePart|addr_vkh|]
  159                         , [humanReadablePart|stake_vkh|]
  160                         , [humanReadablePart|script|]
  161                         ]
  162                     pure bytes
  163                | otherwise ->
  164                     empty
  165         )
  166 
  167     readerBase16 str action = do
  168         bytes <- either (const Nothing) Just . decodeBase16 . encodeUtf8 $ str
  169         action bytes
  170 
  171     readerBase58 str action = do
  172         bytes <- either (const Nothing) Just . decodeBase58 . encodeUtf8 $ str
  173         action bytes
  174 
  175     readerBech32 str action = do
  176         (hrp, dataPart) <- either (const Nothing) Just . Bech32.decodeLenient $ str
  177         bytes <- Bech32.dataPartToBytes dataPart
  178         action bytes hrp

  180 matching :: (Monad f, Alternative f) => Address crypto -> Pattern crypto -> f ()
  181 matching addr = \case
  182     MatchAny (MatchBootstrap matchBootstrap) ->
  183         guard (not (isBootstrap addr) || matchBootstrap)
  184     MatchExact addr' ->
  185         guard (addr' == addr)
  186     MatchPayment payment ->
  187         guard (Just payment == getPaymentPartBytes addr)
  188     MatchDelegation delegation ->
  189         guard (Just delegation == getDelegationPartBytes addr)
  190     MatchPaymentAndDelegation payment delegation -> do
  191         matching addr (MatchPayment payment)
  192         matching addr (MatchDelegation delegation)

  194 data MatchBootstrap
  195     = MatchBootstrap Bool
  196     deriving (Generic, Eq, Show)

  198 pattern IncludingBootstrap :: MatchBootstrap
  199 pattern IncludingBootstrap <- MatchBootstrap True
  200   where
  201     IncludingBootstrap = MatchBootstrap True

  203 pattern OnlyShelley :: MatchBootstrap
  204 pattern OnlyShelley <- MatchBootstrap False
  205   where
  206     OnlyShelley = MatchBootstrap False

  210 data Result crypto = Result
  211     { outputReference :: OutputReference crypto
  212     , address :: Address crypto
  213     , value :: Value crypto
  214     , datumHash :: Maybe (DatumHash crypto)
  215     , point :: Point (Block crypto)
  216     }
  217 deriving instance
  218     ( Show (Point (Block crypto))
  219     ) => Show (Result crypto)
  220 deriving instance
  221     ( Eq (Point (Block crypto))
  222     , Crypto crypto
  223     ) => Eq (Result crypto)

  225 resultToJson
  226     :: forall crypto.
  227         ( CardanoHardForkConstraints crypto
  228         )
  229     => Result crypto
  230     -> Json.Encoding
  231 resultToJson Result{..} = Json.pairs $ mconcat
  232     [ Json.pair "transaction_id"
  233         (transactionIdToJson (getTransactionId outputReference))
  234     , Json.pair "output_index"
  235         (outputIndexToJson (getOutputIndex outputReference))
  236     , Json.pair "address"
  237         (addressToJson address)
  238     , Json.pair "value"
  239         (valueToJson value)
  240     , Json.pair "datum_hash"
  241         (maybe Json.null_ datumHashToJson datumHash)
  242     , Json.pair "slot_no"
  243         (slotNoToJson (getPointSlotNo point))
  244     , Json.pair "header_hash"
  245         (headerHashToJson (unsafeGetPointHeaderHash point))
  246     ]

  248 -- | Match all outputs in transactions from a block that match any of the given
  249 -- pattern.
  250 --
  251 -- Note that this may yield multiple time the same result if being matched by
  252 -- multiple patterns. This is to facilitate building an index of matches to
  253 -- results.
  254 matchBlock
  255     :: forall crypto result.
  256         ( PraosCrypto crypto
  257         , HasHeader (Block crypto)
  258         )
  259     => (Result crypto -> result)
  260     -> [Pattern crypto]
  261     -> Block crypto
  262     -> [result]
  263 matchBlock transform ms blk =
  264     let pt = getPoint blk in foldBlock (fn pt) [] blk
  265   where
  266     fn :: Point (Block crypto) -> Transaction crypto -> [result] -> [result]
  267     fn pt tx results =
  268         concatMap (flip mapMaybeOutputs tx . match pt) ms ++ results
  269 
  270     match
  271         :: Point (Block crypto)
  272         -> Pattern crypto
  273         -> OutputReference crypto
  274         -> Output crypto
  275         -> Maybe result
  276     match pt m outputReference out = do
  277         getAddress out `matching` m
  278         pure $ transform Result
  279             { outputReference
  280             , address = getAddress out
  281             , value = getValue out
  282             , datumHash = getDatumHash out
  283             , point = pt
  284             }
