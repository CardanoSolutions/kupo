   70 -- | Acquire and configure multiple tracers which outputs structured JSON on the
   71 -- standard output. The tracer is concurrent-safe but none buffered, while it is
   72 -- okay for a vast majority of applications, it also relies on a simple
   73 -- implementation and does not perform any caching or hardcore optimizations;
   74 -- For example timestamps are computed on-the-fly for every log messages.
   75 --
   76 withTracers
   77     :: forall tracers. (IsRecordOfTracers tracers IO)
   78     => Handle
   79         -- ^ Handle to which forward logs
   80     -> AppVersion
   81         -- ^ Extra information to embed in the logging envelope.
   82     -> tracers IO 'MinSeverities
   83         -- ^ A configuration of tracers.
   84     -> (tracers IO 'Concrete -> IO ())
   85         -- ^ Callback with acquired and configured tracers.
   86     -> IO ()
   87 withTracers h version tracers action = do
   88     hSetBuffering h LineBuffering
   89     hSetEncoding h utf8
   90     lock <- newTMVarIO ()
   91     action (configureTracers tracers (tracer lock))
   92   where
   93     tracer lock = Tracer $ \(SomeMsg minSeverity tracerName msg) -> do
   94         let severity = getSeverityAnnotation msg
   95         when (severity >= minSeverity) $ liftIO $ withTMVar lock $ \() -> do
   96             mkEnvelop msg severity tracerName >>=
   97                 liftIO . BL8.hPutStrLn h . encodingToLazyByteString
   98 
   99     mkEnvelop
  100         :: forall m msg.
  101             ( ToJSON msg
  102             , MonadIO m
  103             )
  104         => msg
  105         -> Severity
  106         -> String
  107         -> m Encoding
  108     mkEnvelop msg severity tracerName = do
  109         let context = toText (dropWhile isLower tracerName)
  110         timestamp <- liftIO getCurrentTime
  111         threadId <- drop 9 . show <$> liftIO myThreadId
  112         pure $ pairs $ mempty
  113             <> pair "severity"  (toEncoding severity)
  114             <> pair "timestamp" (toEncoding timestamp)
  115             <> pair "thread"    (toEncoding threadId)
  116             <> pair "message"   (pairs $ pair context (toEncoding msg))
  117             <> pair "version"   (toEncoding version)
