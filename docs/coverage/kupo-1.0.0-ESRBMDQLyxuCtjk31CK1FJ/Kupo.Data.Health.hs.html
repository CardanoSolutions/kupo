    5 {-# LANGUAGE RecordWildCards #-}
    6 
    7 module Kupo.Data.Health
    8     ( -- * Health
    9       Health (..)
   10     , emptyHealth
   11 
   12       -- * ConnectionStatus
   13     , ConnectionStatus (..)
   14     ) where
   15 
   16 import Kupo.Prelude
   17 
   18 import Kupo.Data.Cardano
   19     ( SlotNo, slotNoToJson )
   20 
   21 import qualified Data.Aeson.Encoding as Json
   22 
   23 -- | Information about the overall state of the application.
   24 data Health = Health
   25     { connectionStatus :: !ConnectionStatus
   26         -- ^ Condition of the connection with the underlying node.
   27     , mostRecentCheckpoint :: !(Maybe SlotNo)
   28         -- ^ Absolute slot number of the most recent database checkpoint.
   29     , mostRecentNodeTip :: !(Maybe SlotNo)
   30         -- ^ Absolute slot number of the tip of the node
   31     } deriving stock (Generic, Eq, Show)
   32 
   33 instance ToJSON Health where
   34     toEncoding Health{..} = Json.pairs $ mconcat
   35         [ Json.pair
   36             "connection_status"
   37             (toEncoding connectionStatus)
   38         , Json.pair
   39             "most_recent_checkpoint"
   40             (maybe Json.null_ slotNoToJson mostRecentCheckpoint)
   41         , Json.pair
   42             "most_recent_node_tip"
   43             (maybe Json.null_ slotNoToJson mostRecentNodeTip)
   44         ]
   45 
   46 emptyHealth :: Health
   47 emptyHealth = Health
   48     { connectionStatus = Disconnected
   49     , mostRecentCheckpoint = Nothing
   50     , mostRecentNodeTip = Nothing
   51     }
   52 
   53 -- | Reflect the current state of the connection with the underlying node.
   54 data ConnectionStatus
   55     = Connected
   56     | Disconnected
   57     deriving stock (Generic, Eq, Show, Enum, Bounded)
   58 
   59 instance ToJSON ConnectionStatus where
   60     toEncoding = \case
   61         Connected -> Json.text "connected"
   62         Disconnected -> Json.text "disconnected"
