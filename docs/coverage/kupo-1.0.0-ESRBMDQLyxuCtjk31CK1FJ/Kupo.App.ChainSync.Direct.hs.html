   30 -- | A simple pipeline chain-sync clients which offers maximum pipelining and
   31 -- defer handling of requests to callbacks.
   32 mkChainSyncClient
   33     :: forall m block.
   34         ( MonadThrow m
   35         )
   36     => ChainSyncHandler m (Tip block) (Point block) block
   37     -> [Point block]
   38     -> ChainSyncClientPipelined block (Point block) (Tip block) m ()
   39 mkChainSyncClient ChainSyncHandler{onRollBackward, onRollForward} pts =
   40     ChainSyncClientPipelined (pure $ SendMsgFindIntersect pts clientStIntersect)
   41   where
   42     clientStIntersect
   43         :: ClientPipelinedStIntersect block (Point block) (Tip block) m ()
   44     clientStIntersect = ClientPipelinedStIntersect
   45         { recvMsgIntersectFound = \_point _tip -> do
   46             pure $ clientStIdle Zero
   47         , recvMsgIntersectNotFound = \(getTipSlotNo -> tip) -> do
   48             let requestedPoints = pointSlot <$> pts
   49             throwIO $ IntersectionNotFoundException{requestedPoints,tip}
   50         }
   51 
   52     clientStIdle
   53         :: forall n. ()
   54         => Nat n
   55         -> ClientPipelinedStIdle n block (Point block) (Tip block) m ()
   56     clientStIdle n = do
   57         SendMsgRequestNextPipelined $ CollectResponse
   58             (guard (natToInt n < maxInFlight) $> pure (clientStIdle $ Succ n))
   59             (clientStNext n)
   60 
   61     clientStNext
   62         :: forall n. ()
   63         => Nat n
   64         -> ClientStNext n block (Point block) (Tip block) m ()
   65     clientStNext n =
   66         ClientStNext
   67             { recvMsgRollForward = \block tip ->
   68                 onRollForward tip block $> clientStIdle n
   69             , recvMsgRollBackward = \point tip ->
   70                 onRollBackward tip point $> clientStIdle n
   71             }
   72 
   73 -- | Maximum pipelining at any given time. No need to go too high here, it only
   74 -- arms performance beyond a certain point.
   75 --
   76 -- TODO: Make this configurable as it depends on available machine's resources.
   77 maxInFlight :: Int
   78 maxInFlight = 100
