   53 --
   54 -- Checkpoint
   55 --
   56 
   57 pointFromRow
   58     :: DB.Checkpoint
   59     -> Point (CardanoBlock StandardCrypto)
   60 pointFromRow row = BlockPoint
   61     (SlotNo (DB.checkpointSlotNo row))
   62     (fromShelleyHash $ fromShortRawHash proxy $ toShort $ DB.checkpointHeaderHash row)
   63   where
   64     proxy = Proxy @(ShelleyBlock (AlonzoEra StandardCrypto))
   65     fromShelleyHash (Ledger.unHashHeader . unShelleyHash -> UnsafeHash h) =
   66         coerce h
   67 
   68 pointToRow
   69     :: HasCallStack
   70     => Point (CardanoBlock StandardCrypto)
   71     -> DB.Checkpoint
   72 pointToRow = \case
   73     GenesisPoint -> error "pointToRow: genesis point."
   74     BlockPoint slotNo headerHash -> DB.Checkpoint
   75         { DB.checkpointHeaderHash = toRawHash proxy headerHash
   76         , DB.checkpointSlotNo = unSlotNo slotNo
   77         }
   78   where
   79     proxy = Proxy @(CardanoBlock StandardCrypto)
   80 
   81 --
   82 -- Result
   83 --
   84 
   85 resultFromRow
   86     :: HasCallStack
   87     => DB.Input
   88     -> Result
   89 resultFromRow row = Result
   90     { outputReference = unsafeDeserialize' (DB.outputReference row)
   91     , address = (unsafeAddressFromBytes . unsafeDecodeBase16)  (DB.address row)
   92     , value = unsafeDeserialize' (DB.value row)
   93     , datumHash = unsafeDeserialize' <$> (DB.datumHash row)
   94     , point = pointFromRow (DB.Checkpoint (DB.headerHash row) (DB.slotNo row))
   95     }
   96   where
   97     unsafeAddressFromBytes =
   98         fromMaybe (error "unsafeAddressFromBytes") . Ledger.deserialiseAddr
   99 
  100 resultToRow
  101     :: Result
  102     -> DB.Input
  103 resultToRow Result{..} = DB.Input
  104     { DB.outputReference = serialize' outputReference
  105     , DB.address = encodeBase16 (Ledger.serialiseAddr address)
  106     , DB.value = serialize' value
  107     , DB.datumHash = serialize' <$> datumHash
  108     , DB.headerHash = checkpointHeaderHash
  109     , DB.slotNo = checkpointSlotNo
  110     }
  111   where
  112     DB.Checkpoint
  113         { checkpointHeaderHash
  114         , checkpointSlotNo
  115         } = pointToRow point

  121 patternToQueryLike :: Pattern -> Text
  122 patternToQueryLike = \case
  123     MatchAny IncludingBootstrap ->
  124         "LIKE '%'"
  125     MatchAny OnlyShelley ->
  126         "NOT LIKE '8%'"
  127     MatchExact addr ->
  128         "= '" <> encodeBase16 (Ledger.serialiseAddr addr) <> "'"
  129     MatchPayment payment ->
  130         "LIKE '__" <> encodeBase16 payment <> "%'"
  131     MatchDelegation delegation ->
  132         "LIKE '%" <> encodeBase16 delegation <> "' AND len > 58"
  133     MatchPaymentAndDelegation payment delegation ->
  134         "LIKE '__" <> encodeBase16 payment <> encodeBase16 delegation <> "'"
