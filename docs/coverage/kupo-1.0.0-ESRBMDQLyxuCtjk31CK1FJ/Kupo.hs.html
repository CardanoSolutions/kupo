   94 -- | Application entry point.
   95 kupo :: Tracers -> Kupo ()
   96 kupo tr@Tracers{tracerChainSync, tracerHttp, tracerDatabase} = hijackSigTerm *> do
   97     env@Env { health
   98         , configuration = Configuration
   99             { serverHost
  100             , serverPort
  101             , workDir
  102             , since
  103             , patterns
  104             }
  105         } <- ask
  106 
  107      -- 43200 slots = k/f slots
  108      --
  109      -- TODO: k and f should be pulled from the protocol parameters (e.g. via
  110      -- local-state-query).
  111      --
  112      -- Otherwise, the current value may start to be insufficient if the values
  113      -- of `k` or `f` change in the future. In practice,
  114      --
  115      -- (a) there's very little chance that it will change significantly;
  116      -- (b) this kind of change are announced upfront days, if not weeks / months before
  117      --
  118      -- Arbitrarily larger values could be used to cope with future changes
  119      -- since the only impact is on the 'checkpoints' table in the database. We
  120      -- use this value to trim old checkpoints that are not needed for
  121      -- re-establishing an intersection. Since checkpoints are small, storing
  122      -- more is cheap.
  123     let longestRollback = 43200
  124     let dbFile = case workDir of
  125             Dir dir  -> dir </> "kupo.sqlite3"
  126             InMemory -> ":memory:"
  127 
  128     lock <- liftIO newLock
  129     liftIO $ withDatabase tracerDatabase Write lock longestRollback dbFile $ \db -> do
  130         checkpoints <- startOrResume tr db since
  131         withChainProducer tr env db $ \mailbox runChainSyncClient -> do
  132             concurrently3
  133                 -- HTTP Server
  134                 ( runServer
  135                     tracerHttp
  136                     -- NOTE: This should / could probably use a resource pool to
  137                     -- avoid re-creating a new connection on every requests. This is
  138                     -- however pretty cheap with SQLite anyway and the HTTP server
  139                     -- isn't meant to be a public-facing web server serving millions
  140                     -- of clients.
  141                     (withDatabase nullTracer ReadOnly lock longestRollback dbFile)
  142                     (readHealth health)
  143                     serverHost
  144                     serverPort
  145                 )
  146 
  147                 -- Block consumer fueling the database
  148                 ( consumer
  149                     tracerChainSync
  150                     (recordCheckpoint health)
  151                     mailbox
  152                     patterns
  153                     db
  154                 )
  155 
  156                 -- Chain-sync client, fetching blocks from the network
  157                 ( runChainSyncClient
  158                     checkpoints
  159                 )
  160 
  161 
  162 -- TODO: Make the mailbox's size configurable? Larger sizes means larger
  163 -- memory usage at the benefits of slightly faster sync time... Though I
  164 -- am not sure it's worth enough to bother users with this. Between 100
  165 -- and 1000, the heap increases from ~150MB to ~1GB, for a 5-10%
  166 -- synchronization time decrease.
  167 withChainProducer
  168     :: Tracers
  169     -> Env Kupo
  170     -> Database IO
  171     -> (forall block. IsBlock block => Mailbox IO (Tip Block, block) -> ([Point Block] -> IO ()) -> IO ())
  172     -> IO ()
  173 withChainProducer Tracers{tracerChainSync, tracerConfiguration} env db callback = do
  174     case chainProducer of
  175         Ogmios{ogmiosHost, ogmiosPort} -> do
  176             logWith tracerConfiguration ConfigurationOgmios{ogmiosHost, ogmiosPort}
  177             mailbox <- atomically (newMailbox 100)
  178             let chainSyncHandler = producer tracerChainSync (recordCheckpoint health) mailbox db
  179             callback mailbox $ \checkpoints ->
  180                 Ogmios.connect (connectionStatusToggle health) ogmiosHost ogmiosPort $
  181                     Ogmios.runChainSyncClient chainSyncHandler checkpoints
  182 
  183         CardanoNode{nodeSocket, nodeConfig} -> do
  184             logWith tracerConfiguration ConfigurationCardanoNode{nodeSocket,nodeConfig}
  185             mailbox <- atomically (newMailbox 100)
  186             let chainSyncHandler = producer tracerChainSync (recordCheckpoint health) mailbox db
  187             network@NetworkParameters
  188                 { networkMagic
  189                 , slotsPerEpoch
  190                 } <- parseNetworkParameters nodeConfig
  191             logWith tracerConfiguration (ConfigurationNetwork network)
  192             callback mailbox $ \checkpoints ->
  193                 withChainSyncServer
  194                   tracerChainSync
  195                   (connectionStatusToggle health)
  196                   [ NodeToClientV_9 .. NodeToClientV_12 ]
  197                   networkMagic
  198                   slotsPerEpoch
  199                   nodeSocket
  200                   (Direct.mkChainSyncClient chainSyncHandler checkpoints)
  201                   & handle (\e@IntersectionNotFoundException{requestedPoints = points} -> do
  202                       logWith tracerChainSync ChainSyncIntersectionNotFound{points}
  203                       throwIO e
  204                     )
  205   where
  206     Env { health
  207         , configuration = Configuration { chainProducer }
  208         } = env

  214 -- | Application runner with an instantiated environment. See 'newEnvironment'.
  215 runWith :: forall a. Kupo a -> Env Kupo -> IO a
  216 runWith app = runReaderT (unKupo app)
  217 
  218 data Env (m :: Type -> Type) = Env
  219     { configuration :: !Configuration
  220     , health :: !(TVar IO Health)
  221     } deriving stock (Generic)
  222 
  223 newEnvironment
  224     :: Configuration
  225     -> IO (Env Kupo)
  226 newEnvironment configuration = do
  227     health <- newTVarIO emptyHealth
  228     pure Env{configuration, health}
