   38 runChainSyncClient
   39     :: forall m.
   40         ( MonadIO m
   41         , MonadThrow m
   42         )
   43     => ChainSyncHandler m (Tip Block) (Point Block) PartialBlock
   44     -> [Point Block]
   45     -> WS.Connection
   46     -> m ()
   47 runChainSyncClient ChainSyncHandler{onRollBackward, onRollForward} pts ws = do
   48     sendJson ws (encodeFindIntersect pts)
   49     receiveJson ws (decodeFindIntersectResponse pts) >>= \case
   50         Left notFound -> throwIO notFound
   51         Right () -> pure ()
   52     -- NOTE: burst the server with some initial requests, to leverage pipelining.
   53     replicateM_ 100 (sendJson ws encodeRequestNext)
   54     forever $ do
   55         receiveJson ws decodeRequestNextResponse >>= \case
   56             RollBackward tip point -> do
   57                 onRollBackward tip point
   58             RollForward tip block -> do
   59                 onRollForward tip block
   60         sendJson ws encodeRequestNext

   62 -- Connection

   64 connect
   65     :: ConnectionStatusToggle IO
   66     -> String
   67     -> Int
   68     -> (WS.Connection -> IO ())
   69     -> IO ()
   70 connect ConnectionStatusToggle{toggleConnected, toggleDisconnected} host port action =
   71     retry $ WS.runClient host port "/" (\ws -> toggleConnected >> action ws)
   72   where
   73     retry io = catch io $ \e ->
   74         if isRetryableIOException e then do
   75             toggleDisconnected
   76             threadDelay 0.5
   77             retry io
   78         else
   79             throwIO e

   90 data MalformedOrUnexpectedResponseException =
   91     MalformedOrUnexpectedResponse LByteString Json.JSONPath String
   92     deriving (Show)
   93 instance Exception MalformedOrUnexpectedResponseException

   95 sendJson
   96     :: forall m.
   97         ( MonadIO m
   98         )
   99     => WS.Connection
  100     -> Json.Encoding
  101     -> m ()
  102 sendJson ws =
  103     liftIO . WS.sendTextData ws . Json.encodingToLazyByteString
  104 
  105 receiveJson
  106     :: forall m a.
  107         ( MonadThrow m
  108         , MonadIO m
  109         )
  110     => WS.Connection
  111     -> (Json.Value -> Json.Parser a)
  112     -> m a
  113 receiveJson ws decoder =  do
  114     bytes <- liftIO (WS.receiveData ws)
  115     either
  116         (\(path, err) -> throwIO $ MalformedOrUnexpectedResponse bytes path err)
  117         pure
  118         (Json.eitherDecodeWith Json.jsonEOF (Json.iparse decoder) bytes)
