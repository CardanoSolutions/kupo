  218     getPoint
  219         :: Block
  220         -> Point Block
  221     getPoint =
  222         blockPoint
  223 
  224     foldBlock
  225         :: (Transaction -> result -> result)
  226         -> result
  227         -> Block
  228         -> result
  229     foldBlock fn result = \case
  230         BlockByron blk ->
  231             let ignoreProtocolTxs = \case
  232                     ByronTx txId (Ledger.Byron.taTx -> tx) ->
  233                         fn (TransactionByron tx txId)
  234                     _ ->
  235                         identity
  236              in foldr ignoreProtocolTxs result (extractTxs blk)
  237         BlockShelley (ShelleyBlock (Ledger.Block _ txs) _) ->
  238             foldr (fn . TransactionShelley) result (Ledger.Shelley.txSeqTxns' txs)
  239         BlockAllegra (ShelleyBlock (Ledger.Block _ txs) _) ->
  240             foldr (fn . TransactionAllegra) result (Ledger.Shelley.txSeqTxns' txs)
  241         BlockMary (ShelleyBlock (Ledger.Block _ txs) _) ->
  242             foldr (fn . TransactionMary) result (Ledger.Shelley.txSeqTxns' txs)
  243         BlockAlonzo (ShelleyBlock (Ledger.Block _ txs) _) ->
  244             foldr (fn . TransactionAlonzo) result (Ledger.Alonzo.txSeqTxns txs)
  245 
  246     mapMaybeOutputs
  247         :: forall result. ()
  248         => (OutputReference -> Output -> Maybe result)
  249         -> Transaction
  250         -> [result]
  251     mapMaybeOutputs fn = \case
  252         TransactionByron tx (Ledger.Byron.hashToBytes -> bytes) ->
  253             let
  254                 txId = Ledger.TxId $ Ledger.unsafeMakeSafeHash $ UnsafeHash $ toShort bytes
  255                 out :| outs = Ledger.Byron.txOutputs tx
  256              in
  257                 traverseAndTransformByron fromByronOutput txId 0 (out : outs)
  258         TransactionShelley tx ->
  259             let
  260                 body = Ledger.Shelley.body tx
  261                 txId = Ledger.txid @(ShelleyEra StandardCrypto) body
  262                 outs = Ledger.Shelley._outputs body
  263              in
  264                 traverseAndTransform (fromShelleyOutput inject) txId 0 outs
  265         TransactionAllegra tx ->
  266             let
  267                 body = Ledger.Shelley.body tx
  268                 txId = Ledger.txid @(AllegraEra StandardCrypto) body
  269                 outs = Ledger.MaryAllegra.outputs' body
  270              in
  271                 traverseAndTransform (fromShelleyOutput inject) txId 0 outs
  272         TransactionMary tx ->
  273             let
  274                 body = Ledger.Shelley.body tx
  275                 txId = Ledger.txid @(MaryEra StandardCrypto) body
  276                 outs = Ledger.MaryAllegra.outputs' body
  277              in
  278                 traverseAndTransform (fromShelleyOutput identity) txId 0 outs
  279         TransactionAlonzo tx ->
  280             let
  281                 body = Ledger.Alonzo.body tx
  282                 txId = Ledger.txid @(AlonzoEra StandardCrypto) body
  283                 outs = Ledger.Alonzo.outputs' body
  284              in
  285                 traverseAndTransform identity txId 0 outs
  286       where
  287         traverseAndTransformByron
  288             :: forall output. ()
  289             => (output -> Output)
  290             -> TransactionId
  291             -> Natural
  292             -> [output]
  293             -> [result]
  294         traverseAndTransformByron transform txId ix = \case
  295             [] -> []
  296             (out:rest) ->
  297                 let
  298                     outputRef = Ledger.TxIn txId ix
  299                     results   = traverseAndTransformByron transform txId (succ ix) rest
  300                  in
  301                     case fn outputRef (transform out) of
  302                         Nothing ->
  303                             results
  304                         Just result ->
  305                             result : results
  306 
  307         traverseAndTransform
  308             :: forall output. ()
  309             => (output -> Output)
  310             -> TransactionId
  311             -> Natural
  312             -> StrictSeq output
  313             -> [result]
  314         traverseAndTransform transform txId ix = \case
  315             Empty -> []
  316             output :<| rest ->
  317                 let
  318                     outputRef = Ledger.TxIn txId ix
  319                     results   = traverseAndTransform transform txId (succ ix) rest
  320                  in
  321                     case fn outputRef (transform output) of
  322                         Nothing ->
  323                             results
  324                         Just result ->
  325                             result : results

  527 unsafeValueFromList
  528     :: Integer
  529     -> [(ByteString, ByteString, Integer)]
  530     -> Value
  531 unsafeValueFromList ada assets =
  532     Ledger.valueFromList
  533         ada
  534         [ ( unsafePolicyId pid, unsafeAssetName name, q)
  535         | (pid, name, q) <- assets
  536         ]
  537   where
  538     unsafePolicyId =
  539         Ledger.PolicyID
  540         . Ledger.ScriptHash
  541         . UnsafeHash
  542         . toShort
  543         . sizeInvariant (== (digestSize @Blake2b_224))
  544 
  545     unsafeAssetName =
  546         Ledger.AssetName
  547         . sizeInvariant (<= assetNameMaxLength)

  549 valueToJson :: Value -> Json.Encoding
  550 valueToJson (Ledger.Value coins assets) = Json.pairs $ mconcat
  551     [ Json.pair "coins"  (Json.integer coins)
  552     , Json.pair "assets" (assetsToJson assets)
  553     ]
  554   where
  555     assetsToJson :: Map (Ledger.PolicyID StandardCrypto) (Map Ledger.AssetName Integer) -> Json.Encoding
  556     assetsToJson =
  557         Json.pairs
  558         .
  559         Map.foldrWithKey
  560             (\k v r -> Json.pair (assetIdToText k) (Json.integer v) <> r)
  561             mempty
  562         .
  563         flatten
  564 
  565     flatten :: (Ord k1, Ord k2) => Map k1 (Map k2 a) -> Map (k1, k2) a
  566     flatten = Map.foldrWithKey
  567         (\k inner -> Map.union (Map.mapKeys (k,) inner))
  568         mempty
  569 
  570     assetIdToText :: (Ledger.PolicyID StandardCrypto, Ledger.AssetName) -> Text
  571     assetIdToText (Ledger.PolicyID (Ledger.ScriptHash (UnsafeHash pid)), Ledger.AssetName bytes)
  572         | BS.null bytes = encodeBase16 (fromShort pid)
  573         | otherwise     = encodeBase16 (fromShort pid) <> "." <> encodeBase16 bytes
