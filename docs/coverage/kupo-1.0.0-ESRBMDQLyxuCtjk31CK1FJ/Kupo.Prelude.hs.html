  120 isRetryableIOException :: IOException -> Bool
  121 isRetryableIOException e =
  122     isResourceVanishedError e || isDoesNotExistError e || isTryAgainError e
  123   where
  124     isTryAgainError :: IOException -> Bool
  125     isTryAgainError = isInfixOf "resource exhausted" . show
  126 
  127     isResourceVanishedError :: IOException -> Bool
  128     isResourceVanishedError = isInfixOf "resource vanished" . show
  129 
  130 -- | The runtime does not let the application terminate gracefully when a
  131 -- SIGTERM is received. It does however for SIGINT which allows the application
  132 -- to cleanup sub-processes.
  133 --
  134 -- This function install handlers for SIGTERM and turn them into SIGINT.
  135 hijackSigTerm :: MonadIO m => m ()
  136 hijackSigTerm =
  137     liftIO $ void (installHandler softwareTermination handler empty)
  138   where
  139     handler = CatchOnce (raiseSignal keyboardSignal)

  141 -- | An unsafe version of 'decodeBase16'. Use with caution.
  142 unsafeDecodeBase16 :: HasCallStack => Text -> ByteString
  143 unsafeDecodeBase16 =
  144     either error identity . decodeBase16 . encodeUtf8
  145 
  146 -- | Decode a byte string from 'Base58', re-defining here to align interfaces
  147 -- with base16 & base64.
  148 decodeBase58 :: ByteString -> Either Text ByteString
  149 decodeBase58 =
  150     maybe (Left msg) Right . Base58.decodeBase58 Base58.bitcoinAlphabet
  151   where
  152     msg = "failed to decode Base58-encoded string."
  153 
  154 -- | Encode some byte string to 'Text' as Base58. See note on 'decodeBase58'.
  155 encodeBase58 :: ByteString -> Text
  156 encodeBase58 =
  157     decodeUtf8. Base58.encodeBase58 Base58.bitcoinAlphabet
  158 
  159 -- | Encode a ByteString to JSON as a base16 text
  160 byteStringToJson :: ByteString -> Json.Encoding
  161 byteStringToJson =
  162     Json.text . encodeBase16
  163 
  164 -- | Generic JSON encoding, with pre-defined default options.
  165 defaultGenericToEncoding :: (Generic a, GToJSON' Encoding Zero (Rep a)) => a -> Json.Encoding
  166 defaultGenericToEncoding =
  167     genericToEncoding Json.defaultOptions
