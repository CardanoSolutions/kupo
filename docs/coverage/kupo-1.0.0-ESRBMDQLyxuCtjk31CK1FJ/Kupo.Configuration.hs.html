   94 data NetworkParameters = NetworkParameters
   95     { networkMagic :: !NetworkMagic
   96     , systemStart :: !SystemStart
   97     , slotsPerEpoch :: !EpochSlots
   98     } deriving stock (Generic, Eq, Show)
   99       deriving anyclass (ToJSON)

  101 deriving newtype instance ToJSON EpochSlots
  102 deriving newtype instance ToJSON SystemStart
  103 deriving newtype instance ToJSON NetworkMagic
  104 
  105 instance FromJSON NetworkParameters where
  106     parseJSON = Json.withObject "NetworkParameters" $ \obj -> do
  107         nm <- obj .: "networkMagic"
  108         ss <- obj .: "systemStart" >>= parseISO8601
  109         k  <- obj .: "protocolConsts" >>= Json.withObject "protocolConst" (.: "k")
  110         pure NetworkParameters
  111             { networkMagic =
  112                 NetworkMagic (fromIntegral @Integer nm)
  113             , systemStart =
  114                 SystemStart ss
  115             , slotsPerEpoch  =
  116                 EpochSlots (fromIntegral @Integer $ 10 * k)
  117             }
  118       where
  119         parseISO8601 (toString @Text -> str) =
  120             case iso8601ParseM str of
  121                 Nothing -> fail "couldn't parse ISO-8601 date-time."
  122                 Just t  -> pure t
  123 
  124 -- | Construct a 'SystemStart' from a number of seconds.
  125 mkSystemStart :: Int -> SystemStart
  126 mkSystemStart =
  127     SystemStart . posixSecondsToUTCTime . toPicoResolution . toEnum
  128   where
  129     toPicoResolution = (*1000000000000)

  131 -- | Parse a 'Point' from a text string. This alternatively tries two patterns:
  132 --
  133 -- - "origin"        → for a points that refers to the beginning of the blockchain
  134 --
  135 -- - "N.hhhh...hhhh" → A dot-separated integer and base16-encoded digest, which
  136 --                     refers to a specific point on chain identified by this
  137 --                     slot number and header hash.
  138 --
  139 pointFromText :: Text -> Maybe (Point Block)
  140 pointFromText txt =
  141     genesisPointFromText <|> blockPointFromText
  142   where
  143     genesisPointFromText = GenesisPoint
  144         <$ guard (T.toLower txt == "origin")
  145 
  146     blockPointFromText = BlockPoint
  147         <$> slotNoFromText slotNo
  148         <*> headerHashFromText (T.drop 1 headerHash)
  149       where
  150         (slotNo, headerHash) = T.breakOn "." (T.strip txt)
  151 
  152 -- | Parse a slot number from a text string.
  153 slotNoFromText :: Text -> Maybe SlotNo
  154 slotNoFromText txt = do
  155     (slotNo, remSlotNo) <- either (const Nothing) Just (T.decimal txt)
  156     guard (T.null remSlotNo)
  157     pure (SlotNo slotNo)
  158 
  159 -- | Deserialise a 'HeaderHash' from a base16-encoded text string.
  160 headerHashFromText
  161     :: Text
  162     -> Maybe (HeaderHash Block)
  163 headerHashFromText =
  164     fmap (OneEraHash . hashToBytesShort) . hashFromTextAsHex @Blake2b_256

  170 data TraceConfiguration where
  171     ConfigurationNetwork
  172         :: { networkParameters :: NetworkParameters }
  173         -> TraceConfiguration
  174     ConfigurationOgmios
  175         :: { ogmiosHost :: String, ogmiosPort :: Int }
  176         -> TraceConfiguration
  177     ConfigurationCardanoNode
  178         :: { nodeSocket :: FilePath, nodeConfig :: FilePath }
  179         -> TraceConfiguration
  180     ConfigurationInvalidOrMissingOption
  181         :: { hint :: Text }
  182         -> TraceConfiguration
  183     deriving stock (Generic, Show)

  185 instance ToJSON TraceConfiguration where
  186     toEncoding =
  187         defaultGenericToEncoding

  189 instance HasSeverityAnnotation TraceConfiguration where

  192         ConfigurationOgmios{} -> Info
  193         ConfigurationCardanoNode{} -> Info
  194         ConfigurationInvalidOrMissingOption{} -> Error
