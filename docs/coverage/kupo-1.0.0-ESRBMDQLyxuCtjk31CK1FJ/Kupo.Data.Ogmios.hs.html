   89 -- RequestNextResponse

   91 data RequestNextResponse
   92     = RollBackward (Tip Block) (Point Block)
   93     | RollForward  (Tip Block) PartialBlock

   96 -- Encoders
   97 
   98 beginWspRequest :: Json.Series
   99 beginWspRequest = mconcat
  100     [ Json.pair "type" (Json.text "jsonwsp/request")
  101     , Json.pair "version" (Json.text "1.0")
  102     , Json.pair "servicename" (Json.text "ogmios")
  103     ]
  104 
  105 encodeFindIntersect :: [Point Block] -> Json.Encoding
  106 encodeFindIntersect pts = Json.pairs $ beginWspRequest
  107     <> Json.pair "methodname" (Json.text "FindIntersect")
  108     <> Json.pair "args" (Json.pairs $ Json.pair "points" (Json.list encodePoint pts))
  109 
  110 encodeRequestNext :: Json.Encoding
  111 encodeRequestNext = Json.pairs $ beginWspRequest
  112     <> Json.pair "methodname" (Json.text "RequestNext")
  113 
  114 encodePoint :: Point Block -> Json.Encoding
  115 encodePoint = \case
  116     GenesisPoint ->
  117         Json.text "origin"
  118     BlockPoint slotNo headerHash ->
  119         Json.pairs $ mconcat
  120             [ Json.pair "slot" (slotNoToJson slotNo)
  121             , Json.pair "hash" (headerHashToJson headerHash)
  122             ]
  123 
  124 -- Decoders
  125 
  126 decodeFindIntersectResponse
  127     :: [Point Block]
  128     -> Json.Value
  129     -> Json.Parser (Either IntersectionNotFoundException ())
  130 decodeFindIntersectResponse (fmap pointSlot -> requestedPoints) json =
  131     decodeIntersectionFound json <|> decodeIntersectionNotFound json
  132   where
  133     decodeIntersectionFound = Json.withObject "FindIntersectResponse" $ \o -> do
  134         (_ :: Json.Value) <- o .: "result" >>= (.: "IntersectionFound")
  135         return (Right ())
  136 
  137     decodeIntersectionNotFound = Json.withObject "FindIntersectResponse" $ \o -> do
  138         tip <- o .: "result" >>= (.: "IntersectionNotFound") >>= (.: "tip") >>= decodeSlotNoOrOrigin
  139         return (Left IntersectionNotFoundException{requestedPoints, tip})
  140 
  141 decodeRequestNextResponse
  142     :: Json.Value
  143     -> Json.Parser RequestNextResponse
  144 decodeRequestNextResponse json =
  145     decodeRollForward json <|> decodeRollBackward json
  146   where
  147     decodeRollForward =  Json.withObject "RequestNextResponse" $ \o -> do
  148         result <- o .: "result" >>= (.: "RollForward")
  149         block  <- result .: "block"
  150         tip    <- result .: "tip" >>= decodeTipOrOrigin
  151         RollForward tip <$> asum
  152             [ decodeByronBlock block
  153             , decodeShelleyBlock block
  154             , decodeAllegraBlock block
  155             , decodeMaryBlock block
  156             , decodeAlonzoBlock block
  157             ]
  158 
  159     decodeRollBackward = Json.withObject "RequestNextResponse" $ \o -> do
  160         result <- o .: "result" >>= (.: "RollBackward")
  161         point  <- result .: "point" >>= decodePointOrOrigin
  162         tip    <- result .: "tip"   >>= decodeTipOrOrigin
  163         return (RollBackward tip point)
  164 
  165 decodeByronBlock
  166     :: Json.Value
  167     -> Json.Parser PartialBlock
  168 decodeByronBlock = Json.withObject "Block[Byron]" $ \o -> do
  169     block <- o .: "byron"
  170     decodeStandardBlock block <|> decodeEpochBoundaryBlock block
  171   where
  172     decodeStandardBlock = Json.withObject "Block[Byron~Standard]" $ \o -> do
  173         txs <- o .: "body" >>= (.: "txPayload")
  174         slot <- o .: "header" >>= (.: "slot")
  175         headerHash <- o .: "hash" >>= decodeOneEraHash
  176         let point = BlockPoint (SlotNo slot) headerHash
  177         PartialBlock point <$> traverse decodePartialTransaction txs
  178 
  179     decodeEpochBoundaryBlock = Json.withObject "Block[Byron~Standard]" $ \o -> do
  180         slot <- o .: "header" >>= (.: "blockHeight")
  181         headerHash <- o .: "hash" >>= decodeOneEraHash
  182         let point = BlockPoint (SlotNo slot) headerHash
  183         pure (PartialBlock point [])
  184 
  185 decodeShelleyBlock
  186     :: Json.Value
  187     -> Json.Parser PartialBlock
  188 decodeShelleyBlock = Json.withObject "ShelleyBlock" $ \o -> do
  189     shelley <- o .: "shelley"
  190     txs <- shelley .: "body"
  191     point <- decodeBlockPoint shelley
  192     PartialBlock point <$> traverse decodePartialTransaction txs

  221 decodeAddress
  222     :: Text
  223     -> Json.Parser Address
  224 decodeAddress txt =
  225     case patternFromText txt of
  226         Just (MatchExact addr) ->
  227             pure addr
  228         _ ->
  229             empty
  230 
  231 decodeBlockPoint
  232     :: Json.Object
  233     -> Json.Parser (Point Block)
  234 decodeBlockPoint o = do
  235     headerHash <- o .: "headerHash" >>= decodeOneEraHash
  236     slot <- o .: "header" >>= (.: "slot")
  237     pure (BlockPoint (SlotNo slot) headerHash)
  238 
  239 decodeHash
  240     :: HashAlgorithm alg
  241     => Json.Value
  242     -> Json.Parser (Hash alg a)
  243 decodeHash =
  244     Json.parseJSON >=> maybe empty pure . hashFromTextAsHex
  245 
  246 decodeOneEraHash
  247     :: Json.Value
  248     -> Json.Parser (OneEraHash (CardanoEras StandardCrypto))
  249 decodeOneEraHash =
  250     fmap (OneEraHash . toShort . hashToBytes) . decodeHash @Blake2b_256
  251 
  252 decodeOutput
  253     :: Json.Value
  254     -> Json.Parser Output
  255 decodeOutput = Json.withObject "Output" $ \o -> do
  256     address <- o .: "address" >>= decodeAddress
  257     value <- o .: "value" >>= decodeValue
  258     datumHash <- o .:? "datum" >>= traverse (fmap unsafeMakeSafeHash . decodeHash @Blake2b_256)
  259     pure (mkOutput address value datumHash)
  260 
  261 decodePartialTransaction
  262     :: Json.Value
  263     -> Json.Parser PartialTransaction
  264 decodePartialTransaction = Json.withObject "PartialTransaction" $ \o -> do
  265     txId <- o .: "id" >>= decodeTransactionId
  266     outs <- traverse decodeOutput =<< (o .: "body" >>= (.: "outputs"))
  267     pure (PartialTransaction (withReferences txId outs))
  268 
  269 decodePointOrOrigin
  270     :: Json.Value
  271     -> Json.Parser (Point Block)
  272 decodePointOrOrigin json =
  273     decodeOrigin json <|> decodePoint json
  274   where
  275     decodeOrigin = Json.withText "PointOrigin" $ \case
  276         txt | txt == "origin" -> pure GenesisPoint
  277         _ -> empty
  278     decodePoint = Json.withObject "PointOrOrigin" $ \o -> do
  279         slot <- o .: "slot"
  280         hash <- o .: "hash" >>= decodeOneEraHash
  281         return $ BlockPoint (SlotNo slot) hash
  282 
  283 decodeSlotNoOrOrigin
  284     :: Json.Value
  285     -> Json.Parser (WithOrigin SlotNo)
  286 decodeSlotNoOrOrigin json =
  287     decodeOrigin json <|> decodeSlotNo json
  288   where
  289     decodeOrigin = Json.withText "SlotNoOrOrigin" $ \case
  290         txt | txt == "origin" -> pure Origin
  291         _ -> empty
  292     decodeSlotNo = Json.withObject "SlotNoOrOrigin" $ \o -> do
  293         At . SlotNo <$> (o .: "slot")
  294 
  295 decodeTipOrOrigin
  296     :: Json.Value
  297     -> Json.Parser (Tip Block)
  298 decodeTipOrOrigin json =
  299     decodeOrigin json <|> decodeTip json
  300   where
  301     decodeOrigin = Json.withText "TipOrOrigin" $ \case
  302         txt | txt == "origin" -> pure TipGenesis
  303         _ -> empty
  304     decodeTip = Json.withObject "TipOrOrigin" $ \o -> do
  305         slot <- o .: "slot"
  306         hash <- o .: "hash" >>= decodeOneEraHash
  307         blockNo <- o .: "blockNo"
  308         pure $ Tip (SlotNo slot) hash (BlockNo blockNo)
  309 
  310 decodeTransactionId
  311     :: Json.Value
  312     -> Json.Parser TransactionId
  313 decodeTransactionId =
  314     fmap transactionIdFromHash . decodeHash @Blake2b_256
  315 
  316 decodeValue
  317     :: Json.Value
  318     -> Json.Parser Value
  319 decodeValue = Json.withObject "Value" $ \o -> do
  320     coins <- o .: "coins"
  321     assets <- o .:? "assets" .!= mempty >>= traverse decodeAsset . Map.toList
  322     pure (unsafeValueFromList coins assets)
  323   where
  324     decodeBase16' = decodeBase16 . encodeUtf8
  325 
  326     decodeAsset
  327         :: (Text, Integer)
  328         -> Json.Parser (ByteString, ByteString, Integer)
  329     decodeAsset (assetId, quantity) =
  330         case Text.splitOn "." assetId of
  331             [ decodeBase16' -> Right policyId, decodeBase16' -> Right assetName ] -> do
  332                 pure (policyId, assetName, quantity)
  333             [ decodeBase16' -> Right policyId ] -> do
  334                 pure (policyId, mempty, quantity)
  335             _ ->
  336                 empty
