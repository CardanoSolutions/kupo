   49 --
   50 -- Application Bootstrapping
   51 --
   52 
   53 startOrResume
   54     :: forall m.
   55         ( MonadThrow m
   56         , MonadLog m
   57         )
   58     => Tracers
   59     -> Database m
   60     -> Maybe (Point Block)
   61     -> m [Point Block]
   62 startOrResume Tracers{tracerDatabase, tracerConfiguration} Database{..} since = do
   63     checkpoints <- runTransaction (listCheckpointsDesc pointFromRow)
   64     unless (null checkpoints) $ do
   65         let ws = unSlotNo . getPointSlotNo <$> checkpoints
   66         logWith tracerDatabase (DatabaseFoundCheckpoints ws)
   67     case (since, checkpoints) of
   68         (Nothing, []) -> do
   69             logWith tracerConfiguration $ ConfigurationInvalidOrMissingOption
   70                 "No '--since' provided and no checkpoints found in the \
   71                 \database. An explicit starting point (e.g. 'origin') is \
   72                 \required the first time launching the application."
   73             throwIO NoStartingPointException
   74         (Just point, mostRecentCheckpoint:_) -> do
   75             if getPointSlotNo point > getPointSlotNo mostRecentCheckpoint then do
   76                 logWith tracerConfiguration $ ConfigurationInvalidOrMissingOption
   77                     "The point provided through '--since' is more recent than \
   78                     \any of the known checkpoints and it isn't possible to make \
   79                     \a choice for resuming the application: should synchronization \
   80                     \restart from the latest checkpoint or from the provided \
   81                     \--since point? Please dispel the confusion by either choosing \
   82                     \a different starting point (or none at all) or by using a \
   83                     \fresh new database."
   84                 throwIO ConflictingOptionException
   85             else do
   86                 pure (sortOn (Down . getPointSlotNo) (point : checkpoints))
   87         (Nothing, pts) -> do
   88             pure pts
   89         (Just pt, []) ->
   90             pure [pt]
   91 
   92 data NoStartingPointException = NoStartingPointException deriving (Show)
   93 instance Exception NoStartingPointException
   94 
   95 data ConflictingOptionException = ConflictingOptionException  deriving (Show)
   96 instance Exception ConflictingOptionException

   98 --
   99 -- Producer / Consumer
  100 --
  101 
  102 producer
  103     :: forall m block.
  104         ( MonadSTM m
  105         , MonadLog m
  106         )
  107     => Tracer IO TraceChainSync
  108     -> (Tip Block -> Maybe SlotNo -> m ())
  109     -> Mailbox m (Tip Block, block)
  110     -> Database m
  111     -> ChainSyncHandler m (Tip Block) (Point Block) block
  112 producer tr notifyTip mailbox Database{..} = ChainSyncHandler
  113     { onRollBackward = \tip pt -> do
  114         logWith tr (ChainSyncRollBackward (getPointSlotNo pt))
  115         lastKnownSlot <- runTransaction $ rollbackTo (unSlotNo (getPointSlotNo pt))
  116         notifyTip tip (SlotNo <$> lastKnownSlot)
  117     , onRollForward = \tip blk ->
  118         atomically (putMailbox mailbox (tip, blk))
  119     }
  120 
  121 consumer
  122     :: forall m block.
  123         ( MonadSTM m
  124         , MonadLog m
  125         , Monad (DBTransaction m)
  126         , IsBlock block
  127         )
  128     => Tracer IO TraceChainSync
  129     -> (Tip Block -> Maybe SlotNo -> m ())
  130     -> Mailbox m (Tip Block, block)
  131     -> [Pattern]
  132     -> Database m
  133     -> m ()
  134 consumer tr notifyTip mailbox patterns Database{..} = forever $ do
  135     blks <- atomically (flushMailbox mailbox)
  136     let (lastKnownTip, lastKnownBlk) = last blks
  137     let lastKnownPoint = getPoint lastKnownBlk
  138     let lastKnownSlot = getPointSlotNo lastKnownPoint
  139     let inputs = concatMap (matchBlock resultToRow patterns . snd) blks
  140     logWith tr (ChainSyncRollForward lastKnownSlot (length inputs))
  141     notifyTip lastKnownTip (Just lastKnownSlot)
  142     runTransaction $ do
  143         insertCheckpoint (pointToRow lastKnownPoint)
  144         insertInputs inputs

  152 data Tracers' m (kind :: TracerDefinition) = Tracers
  153     { tracerHttp
  154         :: TracerHKD kind (Tracer m TraceHttpServer)
  155     , tracerDatabase
  156         :: TracerHKD kind (Tracer m TraceDatabase)
  157     , tracerChainSync
  158         :: TracerHKD kind (Tracer m TraceChainSync)
  159     , tracerConfiguration
  160         :: TracerHKD kind (Tracer m TraceConfiguration)
  161     } deriving (Generic)
